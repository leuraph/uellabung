In this exercise, you will evaluate integrals.

\begin{parts}
    \part
    Evaluate the following definite integral:

    \begin{subparts}
        \subpart
        \[ \int_{0}^{1} x^2 \dd x \]
    
        \subpart
        \[ \int_{-\pi}^{\pi} \sin(x) \dd x \]
    
    \end{subparts}
    \part
    Evaluate the following indefinite integral:

    \begin{subparts}
        \subpart
        \[ \int 2xe^{x^2} \dd x \]
    \end{subparts}
\end{parts}

\begin{solution}
    \begin{parts}
        \part
        \begin{subparts}
            \subpart
            \[ \int_{0}^{1} x^2 \dd x = \frac{1}{3} \]
            \subpart
            \[ \int_{-\pi}^{\pi} \sin(x) \dd x = 0 \]
        \end{subparts}
        \part
        \begin{subparts}
            \subpart
            \[ \int 2xe^{x^2} \dd x = e^{x^2} + C \]
        \end{subparts}
    \end{parts}
\end{solution}
